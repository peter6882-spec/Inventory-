    // 2. 按鈕動作：加減並寫回雲端

    function handleStockAction(index, actionType) {

        const inputElem = document.getElementById(`input-${index}`);

        const val = parseFloat(inputElem.value);

        if (isNaN(val) || val <= 0) return alert("請輸入大於 0 的數量！");

        

        const item = inventoryData[index];

        let logText = "";


        if (actionType === 'transfer') {

            if (item.freezerStock < val) return alert("大倉總庫存量不足！");

            item.freezerStock = parseFloat((item.freezerStock - val).toFixed(3));

            item.chillerStock = parseFloat((item.chillerStock + val).toFixed(3));

            logText = `🔄 移出備用：【${item.name}】移至現場 +${val} (大倉剩餘: ${item.freezerStock})`;

        } else if (actionType === 'consume') {

            if (item.chillerStock < val) return alert("現場存量不足以扣除！");

            item.chillerStock = parseFloat((item.chillerStock - val).toFixed(3));

            logText = `📤 現場消耗：【${item.name}】扣除 -${val} (現場剩餘: ${item.chillerStock})`;

        } else if (actionType === 'supplier') {

            item.freezerStock = parseFloat((item.freezerStock + val).toFixed(3));

            logText = `📦 進貨入庫：【${item.name}】大倉增加 +${val} (大倉最新: ${item.freezerStock})`;

        }


        // 即時寫回 Firebase

        db.ref('inventory/' + index).update({

            freezerStock: item.freezerStock,

            chillerStock: item.chillerStock

        });


        // 寫入日誌

        const timeStr = new Date().toLocaleTimeString('zh-TW', {hour12: false});

        db.ref('logs').push({ time: timeStr, text: logText });


        inputElem.value = '';

    }

    function renderTable() {

        let lowItemsHtml = '';

                    <td>

                        <div class="action-group">
